function handleTicket(userMessage):

  # 1. Classify the ticket
  requestType  = classify(userMessage, labels=["faq", "billing", "bug", "fraud",
                           "account_access", "permissions", "assessment", "other"])
  productArea  = classify(userMessage, labels=["auth", "payments", "notifications",
                           "data", "admin", "other"])
  urgency      = classify(userMessage, labels=["low", "medium", "high", "critical"])

  # 2. Assess risk
  riskFlags    = detectRisk(userMessage, {
                   hasPII: containsPII(userMessage),
                   isFraudSignal: fraudHeuristics(userMessage),
                   isBillingDispute: requestType == "billing",
                   requiresPermissions: requestType in ["permissions", "account_access"]
                 })

  # 3. Routing decision
  if riskFlags.any(critical=True) or urgency == "critical":
    return escalateToHuman(ticket, reason=riskFlags)

  # 4. Retrieve grounding documents
  relevantDocs = corpusSearch(query=userMessage, productArea=productArea,
                              requestType=requestType, topK=5)

  # 5. Groundedness check
  if relevantDocs.isEmpty() or relevantDocs.maxScore < CONFIDENCE_THRESHOLD:
    return escalateOrDecline(ticket, reason="insufficient_grounding")

  # 6. Generate response
  response = llm.generate(
    systemPrompt = buildSystemPrompt(requestType, productArea, urgency),
    context      = relevantDocs,
    userMessage  = userMessage
  )

  # 7. Output + audit log
  send(response, to=user)
  auditLog.write({ ticket, classification, riskFlags, docsUsed, response })

  # 8. Collect feedback
  feedback = collectCSAT(ticket)
  updateCorpus(feedback)
Key design decisions to think through:

The classification layer is best run as parallel LLM calls (request type, product area, urgency in one prompt or three lightweight calls) rather than chained, to reduce latency.

The escalation gates appear at two points: after risk assessment (pre-retrieval) and after the groundedness check (post-retrieval). The first gate catches high-risk signals before any LLM cost is incurred. The second gate catches cases where the corpus simply doesn't cover the issue.

The audit log is non-negotiable — every routing decision, document used, and response generated should be traceable, especially for billing and fraud cases.

import json
import re
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

# ---------------------------
# Load Corpus
# ---------------------------
with open("corpus/docs.json", "r") as f:
    corpus = json.load(f)

documents = [doc["content"] for doc in corpus]

vectorizer = TfidfVectorizer()
doc_vectors = vectorizer.fit_transform(documents)

# ---------------------------
# Input Layer
# ---------------------------
def preprocess_query(query: str) -> str:
    query = query.lower().strip()
    query = re.sub(r"[^\w\s]", "", query)
    return query

# ---------------------------
# Understanding Layer
# ---------------------------
def classify_intent(query: str) -> str:
    if "refund" in query or "charged" in query:
        return "billing"
    elif "login" in query or "password" in query:
        return "account_access"
    elif "error" in query or "bug" in query:
        return "bug"
    return "faq"

def classify_product_area(query: str) -> str:
    if "login" in query or "password" in query:
        return "login"
    elif "payment" in query or "charged" in query:
        return "billing"
    return "general"

# ---------------------------
# Risk Layer
# ---------------------------
def assess_risk(query: str) -> str:
    if any(word in query for word in ["fraud", "hacked", "unauthorized"]):
        return "HIGH"
    return "LOW"

def assess_urgency(query: str) -> str:
    if "urgent" in query or "immediately" in query:
        return "HIGH"
    return "LOW"

# ---------------------------
# Retrieval Layer
# ---------------------------
def retrieve_documents(query: str, top_k=2):
    query_vec = vectorizer.transform([query])
    scores = cosine_similarity(query_vec, doc_vectors)[0]

    ranked_indices = scores.argsort()[::-1][:top_k]
    results = [corpus[i] for i in ranked_indices if scores[i] > 0]

    return results

# ---------------------------
# Decision Layer
# ---------------------------
def should_escalate(intent, risk, docs):
    if risk == "HIGH":
        return True
    if intent in ["billing", "account_access"]:
        return True
    if len(docs) == 0:
        return True
    return False

# ---------------------------
# Generation Layer
# ---------------------------
def generate_response(query, docs):
    context = " ".join([doc["content"] for doc in docs])

    # Simple grounded response
    return f"Based on our support docs: {context}"

def validate_grounding(response, docs):
    # Basic check: ensure response contains doc content
    return any(doc["content"] in response for doc in docs)

# ---------------------------
# Escalation Layer
# ---------------------------
def create_ticket(query, intent, product_area, risk, urgency):
    return {
        "query": query,
        "intent": intent,
        "product_area": product_area,
        "risk": risk,
        "urgency": urgency
    }

def route_to_human(ticket):
    return f"⚠️ Escalated to human support: {ticket}"

# ---------------------------
# Orchestrator
# ---------------------------
def handle_query(query):

    query = preprocess_query(query)

    intent = classify_intent(query)
    product_area = classify_product_area(query)

    risk = assess_risk(query)
    urgency = assess_urgency(query)

    docs = retrieve_documents(query)

    if should_escalate(intent, risk, docs):
        ticket = create_ticket(query, intent, product_area, risk, urgency)
        return route_to_human(ticket)

    response = generate_response(query, docs)

    if not validate_grounding(response, docs):
        ticket = create_ticket(query, intent, product_area, risk, urgency)
        return route_to_human(ticket)

    return response

# ---------------------------
# Run Demo
# ---------------------------
if __name__ == "__main__":
    while True:
        user_input = input("\nUser: ")
        if user_input.lower() == "exit":
            break

        result = handle_query(user_input)
        print("Agent:", result)

import pandas as pd

def compare_outputs(ground_truth_path, predicted_path):

    gt = pd.read_csv(ground_truth_path)
    pred = pd.read_csv(predicted_path)

    # Merge on ID
    df = pd.merge(gt, pred, on="id", suffixes=("_gt", "_pred"))

    total = len(df)

    # Accuracy calculations
    intent_acc = (df["intent_gt"] == df["intent_pred"]).sum() / total
    product_acc = (df["product_area_gt"] == df["product_area_pred"]).sum() / total
    decision_acc = (df["decision_gt"] == df["decision_pred"]).sum() / total

    # Exact match (all fields correct)
    exact_match = (
        (df["intent_gt"] == df["intent_pred"]) &
        (df["product_area_gt"] == df["product_area_pred"]) &
        (df["decision_gt"] == df["decision_pred"])
    ).sum() / total

    # Error breakdown
    errors = df[
        (df["decision_gt"] != df["decision_pred"])
    ][["id", "query_gt", "decision_gt", "decision_pred"]]

    print("\n📊 Evaluation Results:")
    print(f"Total Samples: {total}")
    print(f"Intent Accuracy: {intent_acc:.2f}")
    print(f"Product Area Accuracy: {product_acc:.2f}")
    print(f"Decision Accuracy: {decision_acc:.2f}")
    print(f"Exact Match Accuracy: {exact_match:.2f}")

    print("\n❌ Decision Errors:")
    print(errors)

    return {
        "intent_acc": intent_acc,
        "product_acc": product_acc,
        "decision_acc": decision_acc,
        "exact_match": exact_match
    }

import pandas as pd
from sklearn.metrics import confusion_matrix

def evaluate_system(gt_path, pred_path):

    # Load data
    gt = pd.read_csv(gt_path)
    pred = pd.read_csv(pred_path)

    df = pd.merge(gt, pred, on="id", suffixes=("_gt", "_pred"))

    total = len(df)

    # ---------------------------
    # Basic Accuracy Metrics
    # ---------------------------
    intent_acc = (df["intent_gt"] == df["intent_pred"]).mean()
    product_acc = (df["product_area_gt"] == df["product_area_pred"]).mean()
    decision_acc = (df["decision_gt"] == df["decision_pred"]).mean()

    exact_match = (
        (df["intent_gt"] == df["intent_pred"]) &
        (df["product_area_gt"] == df["product_area_pred"]) &
        (df["decision_gt"] == df["decision_pred"])
    ).mean()

    # ---------------------------
    # Safety Score (CRITICAL)
    # ---------------------------
    safety_score = 0
    for _, row in df.iterrows():
        if row["decision_gt"] == "ESCALATE":
            if row["decision_pred"] == "ESCALATE":
                safety_score += 2   # correct escalation
            else:
                safety_score -= 3   # dangerous mistake
        else:
            if row["decision_pred"] == "REPLY":
                safety_score += 1
            else:
                safety_score -= 1

    safety_score = safety_score / total

    cm = confusion_matrix(df["decision_gt"], df["decision_pred"], labels=["ESCALATE", "REPLY"])

        0.2 * product_acc +

    elif final_score >= 0.75:

    elif final_score >= 0.4:
        rating = "D (Needs Improvement)"
    else:
        rating = "F (Unsafe)"

    # Error Analysis
    # ---------------------------
    decision_errors = df[df["decision_gt"] != df["decision_pred"]]

def compute_risk_score(df):
    score = 0

    for _, row in df.iterrows():
        gt = row["decision_gt"]
        pred = row["decision_pred"]

        if gt == "ESCALATE" and pred == "ESCALATE":
            score += 3   # correct critical decision
        elif gt == "ESCALATE" and pred == "REPLY":
            score -= 5   # VERY BAD (missed escalation)
        elif gt == "REPLY" and pred == "ESCALATE":
            score -= 1   # over-escalation (minor penalty)
        else:
            score += 1   # correct safe reply

    return score / len(df)

def compute_final_score(metrics, risk_score, over_esc, intent_acc, product_acc):

    final = (
        0.35 * metrics["recall_escalate"] +   # MOST important
        0.20 * metrics["precision_escalate"] +
        0.20 * max(0, risk_score) +
        0.15 * intent_acc +
        0.10 * product_acc

    # penalty if system escalates too much
    if over_esc > 0.4:
        final -= 0.1

    return max(0, min(1, final))
